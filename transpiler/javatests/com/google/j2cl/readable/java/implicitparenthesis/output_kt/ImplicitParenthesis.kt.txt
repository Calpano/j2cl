package implicitparenthesis

open class ImplicitParenthesis {
 open fun test() {
  implicitparenthesis.ImplicitParenthesis.examples()
  implicitparenthesis.ImplicitParenthesis.nonExamples()
 }

 companion object {
  @JvmStatic
  private fun examples() {
   var primitiveInt: Int
   var boxedInt: Int?
   if (true) {
    primitiveInt = 5
    boxedInt = Int(10)
    primitiveInt = if (primitiveInt == 5) Int(15) else Int(30)
    assert(primitiveInt == 15)
   }
   if (true) {
    primitiveInt = 5
    boxedInt = Int(10)
    primitiveInt = boxedInt += primitiveInt
    assert(primitiveInt == 15)
   }
   if (true) {
    primitiveInt = 5
    boxedInt = Int(10)
    primitiveInt = boxedInt = primitiveInt
    assert(primitiveInt == 5)
   }
   if (true) {
    primitiveInt = 5
    boxedInt = Int(10)
    primitiveInt = boxedInt++
    assert(primitiveInt == 10)
   }
   if (true) {
    primitiveInt = 5
    boxedInt = Int(10)
    primitiveInt = ++boxedInt
    assert(primitiveInt == 11)
   }
  }

  @JvmStatic
  private fun nonExamples() {
   var primitiveInt: Int
   var boxedInt: Int?
   var boxedBoolean: Boolean?
   if (true) {
    primitiveInt = 5
    boxedInt = Int(10)
    primitiveInt = boxedInt - primitiveInt
    assert(primitiveInt == 5)
   }
   if (true) {
    primitiveInt = 5
    boxedInt = Int(10)
    boxedBoolean = true
    boxedBoolean = boxedInt == primitiveInt
    assert(boxedBoolean == false)
   }
  }

  @JvmStatic
  private fun parenthesisRemoval() {
   Int(1 + 2)
  }
 }
}
