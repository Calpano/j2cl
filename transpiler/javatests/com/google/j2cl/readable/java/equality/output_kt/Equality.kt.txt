package equality

open class Equality {
 open fun test() {
  if (true) {
   assert(false == false)
   assert(0 != 1)
  }
  if (true) {
   assert(Any() != Any())
  }
  if (true) {
   assert(0.0.toDouble() == 0.0.toDouble())
  }
  if (true) {
   assert(0.0f.toFloat() == 0.0f.toFloat())
  }
  if (true) {
   assert(null != Any())
   assert(Any() != null)
  }
 }
}
