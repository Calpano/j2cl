open class SwitchStatement {
 fun main() {
  SWITCH@ do when ("one") {
   "one" ->
   "two" ->
    break@SWITCH
   else ->
    return Unit
  } while (false)
  SWITCH@ do when ('1') {
   '1' ->
   '2' ->
    break@SWITCH
   else ->
    return Unit
  } while (false)
  SWITCH@ do when (1) {
   -2 ->
   1 ->
   2 ->
    break@SWITCH
   else ->
    return Unit
  } while (false)
  SWITCH@ do when (TODO("JavaScriptConstructorReference").ONE) {
   TODO("JavaScriptConstructorReference").ONE ->
   TODO("JavaScriptConstructorReference").TWO ->
    break@SWITCH
   else ->
    return Unit
  } while (false)
 }

 companion object {
  @JvmStatic
  fun testSwitchVariableDeclarations() {
   SWITCH@ do when (3) {
    1 ->
     val unassigned: Int; val unassigned2: Int
     val i: Int = 0
     val j: Int = 2; val b: Int = j + 1
     break@SWITCH
    3 ->
     i = 3
     assert(i == 3)
     return Unit
   } while (false)
   SWITCH@ do when (5) {
    5 ->
     val i: Int = 1
     break@SWITCH
   } while (false)
   assert(false)
  }
 }
}
