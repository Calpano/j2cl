package switchstatement

enum class Numbers: kotlin.Enum<switchstatement.Numbers?> {
 internal constructor(): super()

 companion object {
  @kotlin.jvm.JvmField val ONE: switchstatement.Numbers = switchstatement.Numbers()

  @kotlin.jvm.JvmField val TWO: switchstatement.Numbers = switchstatement.Numbers()
 }
}

open class SwitchStatement {
 constructor(): super()

 open fun main() {
  SWITCH@ do {
   when ("one") {
    "one" ->
    "two" ->
     break@SWITCH
    else ->
     return
   }
  } while (false)
  SWITCH_1@ do {
   when ('1') {
    '1' ->
    '2' ->
     break@SWITCH_1
    else ->
     return
   }
  } while (false)
  SWITCH_2@ do {
   when (1) {
    -2 ->
    1 ->
    2 ->
     break@SWITCH_2
    else ->
     return
   }
  } while (false)
  SWITCH_3@ do {
   when (switchstatement.Numbers.ONE) {
    switchstatement.Numbers.ONE ->
    switchstatement.Numbers.TWO ->
     break@SWITCH_3
    else ->
     return
   }
  } while (false)
 }

 companion object {
  @kotlin.jvm.JvmStatic
  internal fun testSwitchVariableDeclarations() {
   val unassigned: Int
   val unassigned2: Int
   val i: Int
   val j: Int
   val b: Int
   val i_1: Int
   SWITCH@ do {
    when (3) {
     1 ->
      null
      i = 0
      if (true) {
       j = 2
       b = j + 1
      }
      break@SWITCH
     3 ->
      i = 3
      assert(i == 3)
      return
    }
   } while (false)
   SWITCH_1@ do {
    when (5) {
     5 ->
      i_1 = 1
      break@SWITCH_1
    }
   } while (false)
   assert(false)
  }
 }
}
