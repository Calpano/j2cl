package multipleconstructors

open class MultipleConstructors {
 internal var id: Int = 0

 internal var flag: Boolean = false

 constructor(id: Int): super() {
  this.id = id
  this.flag = id == 0
 }

 constructor(flag: Boolean): super() {
  this.id = - 1
  this.flag = flag
 }

 constructor(id: Int, flag: Boolean): super() {
  this.id = id
  this.flag = flag
 }
}
