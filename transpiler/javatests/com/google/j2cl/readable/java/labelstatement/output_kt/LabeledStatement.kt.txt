 open fun simpleStatement() {
  LABEL@ do this.foo() while (false)
  LOOP@ do {
   LABEL_1@ do continue@LOOP while (false)
  } while (false)
  LABEL_2@ do return Unit while (false)
 }

 open fun block() {
  LABEL@ do {
   break@LABEL
  } while (false)
 }

 open fun ifStatement() {
  LABEL@ do if (true) {
   break@LABEL
  } while (false)
 }

 open fun switchStatement() {
  LABEL@ do when (0) {
   0 ->
    if (true) {
     break@LABEL
    }
  } while (false)
 }
