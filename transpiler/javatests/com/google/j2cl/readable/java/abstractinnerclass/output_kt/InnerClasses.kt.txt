package abstractinnerclass

open class InnerClasses {}

interface A {
 fun foo()
}

abstract class B: abstractinnerclass.InnerClasses.A {
 internal open fun bar() {
  this.foo()
 }
}

abstract class C: abstractinnerclass.InnerClasses.A {
 val `this`: abstractinnerclass.InnerClasses?

 internal open fun bar() {
  this.foo()
 }
}
