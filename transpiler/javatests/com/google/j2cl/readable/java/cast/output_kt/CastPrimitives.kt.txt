 open fun testImplicitLongAssignmentCasts() {
  val fbyte: Byte = 11
  val fchar: Char = 12
  val fshort: Short = 13
  val fint: Int = 14
  val flong: Long = 15L
  val ffloat: Float = 16.0f
  val fdouble: Double = 17.0
  var tlong: Long = 0L
  if (true) {
   tlong = fbyte.toLong()
   tlong = flong
  }
  if (true) {
   tlong = fint.toLong()
   tlong = tlong + fint.toLong()
   tlong = tlong << fint
  }
  if (true) {
   tlong = tlong + fchar.code.toLong()
   tlong = tlong + flong
   tlong = (tlong.toFloat() + ffloat).toLong()
  }
  if (true) {
   tlong = flong * fint.toLong()
   tlong = flong >> fint
  }
  if (true) {
   tlong = flong + fshort.toLong()
   tlong = flong + flong
  }
  if (true) {
   tlong = flong << tlong
   tlong = tlong << flong
  }
  if (true) {
   tlong = (0 + 1).toLong() + 2L
  }
 }
